class Solution:
    def hasCycle(self, head: Optional[ListNode]) -> bool:

        #edge case
        if not head:
            return False
        
        if head.next == None:
            return False

        
        viz = set()
        falsehead = head

        while falsehead:
            if falsehead not in viz:
                viz.add(falsehead)
                falsehead = falsehead.next
            else:
                return True 
        
        return False
